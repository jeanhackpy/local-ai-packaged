        commentaires.append(commentaire)

    return commentaires

import requests
from bs4 import BeautifulSoup

def extraire_commentaires_amazon(url):
    # En-têtes pour simuler un navigateur
    headers = {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/58.0.3029.110 Safari/537.3'}

    # Envoyer une requête GET à l'URL
    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.content, 'html.parser')

    commentaires = []

    # Trouver tous les éléments de commentaire sur la page
    elements_commentaire = soup.find_all('div', {'data-hook': 'review'})

    # Parcourir les éléments de commentaire et extraire le texte et la note
    for element in elements_commentaire:
        texte_commentaire = element.find('span', {'data-hook': 'review-body'}).text.strip()
        note_commentaire = element.find('span', {'data-hook': 'review-star-rating'}).text.strip()
        commentaire = {'Texte': texte_commentaire, 'Note': note_commentaire}
        commentaires.append(commentaire)

    return commentaires
